```python
import os
from pathlib import Path
from pdf2image import convert_from_path

def convert_pdfs_to_images(pdf_folder, output_folder, dpi=300):
    """
    –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –≤—Å–µ PDF –≤ –ø–∞–ø–∫–µ –≤ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    
    Args:
        pdf_folder: –ü–∞–ø–∫–∞ —Å PDF —Ñ–∞–π–ª–∞–º–∏
        output_folder: –ü–∞–ø–∫–∞ –¥–ª—è —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è JPG
        dpi: –†–∞–∑—Ä–µ—à–µ–Ω–∏–µ (—Ä–µ–∫–æ–º–µ–Ω–¥—É–µ—Ç—Å—è 300)
    """
    pdf_folder = Path(pdf_folder)
    output_folder = Path(output_folder)
    output_folder.mkdir(exist_ok=True)
    
    for pdf_file in pdf_folder.glob("*.pdf"):
        print(f"–ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è: {pdf_file.name}")
        
        # –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å PDF –≤ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
        images = convert_from_path(str(pdf_file), dpi=dpi)
        
        # –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –∫–∞–∂–¥—É—é —Å—Ç—Ä–∞–Ω–∏—Ü—É
        for i, image in enumerate(images, 1):
            base_name = pdf_file.stem
            output_name = f"{base_name}_page_{i}.jpg"
            output_path = output_folder / output_name
            image.save(output_path, 'JPEG', quality=95)
            print(f"  ‚Üí –°–æ—Ö—Ä–∞–Ω–µ–Ω–æ: {output_name}")

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
if __name__ == "__main__":
    # –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å train PDF
    convert_pdfs_to_images(
        pdf_folder="raw_pdfs/train",
        output_folder="images/train"
    )
    
    # –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å val PDF
    convert_pdfs_to_images(
        pdf_folder="raw_pdfs/val",
        output_folder="images/val"
    )
```
